1.) What is each pokemon's primary type?
SELECT pokemons.name, types.name FROM pokemons JOIN types ON pokemons.primary_type = types.id;

2.) What is Rufflet's secondary type?
SELECT pokemons.name, types.name FROM pokemons JOIN types ON pokemons.Seconadry_type = types.id WHERE pokemons.name = 'Rufflet';

3.) What are the names of the pokemon that belong to the trainer with trainerID 303?
SELECT trainers.trainername AS Trainers_Name,pokemons.name AS Pokemon_Name FROM trainers JOIN pokemons ON pokemons.id = trainers.trainerID WHERE pokemons.id = 303;

4.) How many pokemon have a secondary type Poison
SELECT types.name, COUNT(pokemons.secondary_type) AS EachType FROM pokemons LEFT JOIN types ON pokemons.secondary_type = types.id GROUP BY types.name;

SELECT types.name, COUNT(pokemons.primary_type) AS EachType FROM pokemons LEFT JOIN types ON pokemons.primary_type = types.id GROUP BY types.name;
